using UnityEngine;
// 파일: GreenEntity.cs
public class GreenEntity : AIEntity
{
    protected override bool IsPredator(EntityBase entity)
    {
        // 녹색 개체의 천적은 '파란색 개체'
        return entity is BlueEntity;
    }

    protected override bool IsPrey(EntityBase entity)
    {
        // 녹색 개체의 먹이는 '보라색 개체'
        return entity is PurpleEntity;
    }
    
    // Move, Attack, Die 등의 나머지 추상 메서드도 구현 필요...
    public override void Attack(EntityBase target)
    {
        Debug.Log($"{name}이(가) {target.GetType().Name}을(를) 공격!");
        Destroy(target.gameObject);
    }

    public override void Die()
    {
        Debug.Log($"{name}이(가) 사망했습니다.");
        Destroy(gameObject);
    }

    public override void Move(Vector2 direction)
    {
        transform.Translate(direction.normalized * MoveSpeed * Time.deltaTime);
    }
}
